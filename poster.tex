          \begin{block}{A Timeline of Firsts for Women in Statistics in Canada}
            \begin{timeline}{1948}{2021}{5cm}{5cm}{75cm}{68cm}
            \small  
            \entry{1948}{Universal Declaration of Human Rights}
            %---
            % 1948: Universal Declaration of Human Rights
            % https://www.un.org/en/about-us/universal-declaration-of-human-rights
            %---
            \entry{1956}{The female pay act made wage discrimination based on gender against the law}
            \entry{1960}{Voting rights inclusive of gender, race, and heritage} 
            %---
            % 1960: Voting rights inclusive of gender, race, and heritage (Indian status).
            % https://electionsanddemocracy.ca/voting-rights-through-time-0/brief-history-federal-voting-rights-canada
            %---
            \entry{1967}{The University of Manitoba forms Canada's first Department of Statistics}
            \entry{1969}{\textbf{Isobel Loutit} became chair of the Montreal Section of the American Society for Quality Control (ASQC)}
            \entry{1970}{Royal Commission on the Status of Women detailing discrimination and barriers faced by women across Canada}
            \entry{1971}{Maternity benefits added to unemployment insurance.}
            \entry{1972}{\textbf{Dr. Sylvia Ostry} is the first female Chief Statistician of Canada}
            \entry{1975}{\textbf{Audrey Duthie} and \textbf{Kathleen Kocherlakota} are the first women to be elected to the Board of Directors of the Statistical Science Association of Canada, a predecessor to the Statistical Society of Canada (SSC)}
            \entry{1977}{The Canadian Human Rights Act (CHRA) was passed, prohibiting discrimination on the basis of sex, and ensuring equal pay for work of equal value.}
            \plainentry{1977}{\textbf{Dr. Priscila E. (Cindy) Greenwood} is awarded a grant of \$50,000 from the Peter Wall Institute of Advanced Studies for their first topic of study}
            \entry{1978}{The Canada Labour Code was amended to eliminate pregnancy as a basis for lay-off or dismissal.}
            \entry{1980}{\textbf{Estella Bee Dagum} receives the inaugural award of the Julius Shiskin Memorial Award for Economic Statistics, given by the Business and Economic Statistics Section of the American Statistical Association.}
            \entry{1983}{The CHRA was amended to prohibit sexual harassment and to ban discrimination on the basis of pregnancy and family or marital status.}
            \entry{1986}{The federal Employment Equity Act was introduced, aimed at redressing historic and systemic discrimination of target group populations.}
            \entry{1987}{\textbf{Maureen Tingley} is the first female to win Pierre Robillard award}
            \entry{1990}{\textbf{Dr. Constance van Eeden} is the first woman to receive the Gold medal of the Statistical Society of Canada.}
            \entry{1991}{\textbf{Dr. Agnes Herzberg} is the first female president of the Statistical Society of Canada.}
            \entry{1992}{\textbf{Nancy Reid} is the first statistician working in Canada to receive the COPSS Presidents' Award.}
            \entry{1992}{\textbf{Lise Manchester} is awarded the Canadian Journal of Statistics award or best paper.}
            \entry{1993}{\textbf{K Brenda MacGibbon Taylor} is the first woman to be the Chair of the Statistical Sciences Grant Selection Committee.}
            \entry{1999}{\textbf{Dr. Agnes Herzberg} is the first woman recipient of the SSC Distinguished service award.}
            \entry{2001}{\textbf{Colleen Cutler} is the first woman to be awarded the CRM-SCC Prize.}
            \entry{2002}{\textbf{Jane F. Gentleman} wins the first annual Janet L. Norwood Award for Outstanding Achievement by a Woman in the Statistical Sciences.}
            \entry{2007}{\textbf{Agnes Herzberg} is the first female to be presented with an SSC Honorary award.}
            \entry{2008}{Prime Minister Stephen Harper issues a statement of apology to former students of Residential Schools in Canada for the harm caused by assimilationist goals, abuse, and cultural loss.}
            \entry{2009}{\textbf{Gail Ivanoff} is the first female to be presented with an SSC Honorary award.}
            \entry{2012}{\textbf{Mary Thompson} is the first scientific director of Canadian Statistical Science Institute CANSSI.}
            \entry{2014}{\textbf{Nancy Reid} became the first female Canadian statistician to receive the order of Canada.}
            \entry{2016}{An Inquiry into Missing and Murdered Indigenous Women and Girls is launched in response to calls for action from families, communities, and organizations}
            %---
            % 2016: An Inquiry into Missing and Murdered Indigenous Women and Girls
            % http://education.historicacanada.ca/files/426/Key_Moments_in_Indigenous_History_Timeline.pdf
            % https://www.mmiwg-ffada.ca/
            %---
            \entry{2017}{\textbf{Charmaine Dean} became the first female statistician to become Vice President research of a Canadian university.}
            \end{timeline}
